2. Write a shell script to calculate factorial of ‘n’ numbers.

Program:

clear
echo "Enter how many numbers"
read n
i=1
while [ $i -le $n ]
do 
	fact=1
	x=1
	echo "Enter the number to find the factorial"
	read m
	while [ $x -le $m ]
	do 
		fact=`expr $fact \* $x`
		x=`expr $x + 1`
	done
	echo "Factorial of $m is $fact"
	i=`expr $i + 1`
done
